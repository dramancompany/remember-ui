## Basic usage

<Playground>
  {() => {
    const [isChecked, setIsChecked] = React.useState(true);
    const toggle = () => setIsChecked(!isChecked);
    return (
      <>
        <Container direction="column">
          <div>체크박스 on/off/disabled</div>
          <Container direction="row">
            <Checkbox state="on" />
            <Checkbox state="off" />
            <Checkbox state="on" disabled />
          </Container>
        </Container>
        
        <Container direction="column">
          <div>체크박스 onClick Event Bind</div>
          <Container direction="row">
            <Checkbox state={isChecked ? 'on' : 'off'} onClick={toggle} />
            <Checkbox
              state={isChecked ? 'on' : 'off'}
              onClick={toggle}
              disabled
            />
          </Container>
        </Container>
      </>
    );
  }}
</Playground>
