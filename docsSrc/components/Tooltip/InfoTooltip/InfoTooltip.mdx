## Basic usage

<Playground>
  {() => {
    const [visible, setVisible] = useState(true);
    const onClickToolTip = () => {
      setVisible(false);
    };
    return (
      <DoczContainer>
        <div>ToolTip</div>
        <InfoTooltip
          visible={visible}
          content={<span>툴팁을 클릭하면, 다시 보지 않게 됩니다.</span>}
          onClick={onClickToolTip}
        />
      </DoczContainer>
    );
  }}
</Playground>
